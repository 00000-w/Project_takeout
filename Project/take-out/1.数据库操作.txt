    UNIQUE KEY `idx_name` (`name`)

创建套餐菜品关联表 （多对多关系中间表  一个套餐可以有多个菜品，一个菜品可以在多个套餐内都有）
解析：中间表（也叫关联表 / 连接表） 是专门用来 “拆解” 多对多关系的表 —— 因为数据库无法直接存储多对多关系，必须通过中间表将其拆分为两个一对多关系
CREATE TABLE `setmeal_dish` (
    `id` bigint NOT NULL AUTO_INCREMENT COMMENT '主键',
    `setmeal_id` bigint NOT NULL COMMENT '套餐id',
    `dish_id` bigint NOT NULL COMMENT '菜品id',
    `name` varchar(32) DEFAULT NULL COMMENT '菜品名称（冗余字段）',
    `price` decimal(10, 2) NOT NULL COMMENT '菜品原价（冗余字段）',
    `copies` int NOT NULL COMMENT '份数',
    PRIMARY KEY (`id`),
    UNIQUE KEY `idx_setmeal_dish` (`setmeal_id`, `dish_id`) COMMENT '套餐和菜品唯一索引'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT '套餐菜品关系';

1.订单表
CREATE TABLE `order` (
    `id` bigint NOT NULL AUTO_INCREMENT COMMENT '主键',
    `number` varchar(50) DEFAULT NULL COMMENT '订单号',
    `status` int NOT NULL DEFAULT '1' COMMENT '订单状态 1：待付款 2：待接单 3.已接单 4.派送中 5.已完成 6.已取消 7.退款',
    `user_id` bigint NOT NULL COMMENT '下单用户id',
    `address_book_id` bigint DEFAULT NULL COMMENT '地址id',
    `order_time` datetime NOT NULL COMMENT '下单时间',
    `checkout_time` datetime DEFAULT NULL COMMENT '结账时间',
    `pay_method` int NOT NULL DEFAULT '1' COMMENT '支付方式 1：微信 2：支付宝',
    `pay_status` tinyint NOT NULL DEFAULT '0' COMMENT '支付状态 0：未支付 1：已支付 2：退款',
    `amount` decimal(10, 2) NOT NULL COMMENT '实收金额',
    `remark` varchar(100) DEFAULT NULL COMMENT '备注',
    `phone` varchar(20) DEFAULT NULL COMMENT '手机号',
    `address` varchar(255) DEFAULT NULL COMMENT '地址',
    `user_name` varchar(50) DEFAULT NULL COMMENT '用户名称',
    `consignee` varchar(50) DEFAULT NULL COMMENT '收货人',
    PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT '订单表';
